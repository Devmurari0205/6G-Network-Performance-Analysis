import streamlit as st
import pandas as pd
import numpy as np
import plotly.express as px

# Page setup
st.set_page_config(page_title="6G Dashboard", layout="wide")

st.title("📡 6G Network Performance Dashboard")

# Load data
df = pd.read_csv("network_performance.csv")

# Fix Date
df["Date"] = pd.to_datetime(df["Date"], dayfirst=True, errors='coerce')
df = df.dropna(subset=["Date"])

# Risk Category
def risk_category(row):
    if row["Network_Latency_ms"] > 30 or row["Packet_Loss"] > 5:
        return "Critical"
    elif row["Network_Latency_ms"] > 15:
        return "Warning"
    else:
        return "Normal"

df["Risk_Level"] = df.apply(risk_category, axis=1)

# Sidebar filter
st.sidebar.header("Filters")

operation_mode = st.sidebar.multiselect(
    "Operation Mode",
    df["Operation_Mode"].unique(),
    default=df["Operation_Mode"].unique()
)

df_filtered = df[df["Operation_Mode"].isin(operation_mode)]

# KPI
col1, col2, col3 = st.columns(3)

col1.metric("Avg Latency", round(df_filtered["Network_Latency_ms"].mean(), 2))
col2.metric("Avg Packet Loss", round(df_filtered["Packet_Loss"].mean(), 2))
col3.metric("Avg Speed", round(df_filtered["Production_Speed_units_per_hr"].mean(), 2))

# Charts
st.subheader("Latency vs Speed")
fig1 = px.scatter(df_filtered, x="Network_Latency_ms", y="Production_Speed_units_per_hr", color="Efficiency_Status")
st.plotly_chart(fig1)

st.subheader("Packet Loss vs Error Rate")
fig2 = px.scatter(df_filtered, x="Packet_Loss", y="Error_Rate", color="Risk_Level")
st.plotly_chart(fig2)

st.subheader("Efficiency Distribution")
fig3 = px.histogram(df_filtered, x="Efficiency_Status", color="Efficiency_Status")
st.plotly_chart(fig3)

st.subheader("Latency Trend")
trend = df.groupby("Date")["Network_Latency_ms"].mean().reset_index()
fig4 = px.line(trend, x="Date", y="Network_Latency_ms")
st.plotly_chart(fig4)

st.success("Dashboard running successfully ✅")
